These are notes I made for myself because end-to-end automating the Windows
build is unlikely to happen because Windows itself is not conducive to
automating that sort of thing...  So if you're not me, and you're not
building MusiKernel for Windows, you can safely ignore this file :-)

MinGW{32,64} build:

yumdownloader --source libsndfile fftw portmidi # vorbis-tools
source  "THIS SCRIPT" >>>
PREFIX=x86_64-w64-mingw32
export CC=$PREFIX-gcc
export CXX=$PREFIX-g++
export CPP=$PREFIX-cpp
export RANLIB=$PREFIX-ranlib
#export PATH="$HOME/bin:$PATH"
export PREFIX=/usr
<<<


foreach (package):
	rpmdev-extract [PACKAGE].rpm
	cd [PACKAGE DIR]
	tar xvf [PACKAGE TARBALL]  #except stupid portmidi which is unzip [zip file]
	cd [TARBALL DIR]
	./configure --prefix=/usr --host=x86_64-w64-mingw32
	make
	make DESTDIR=`pwd`/../../BIN

	export CFLAGS="-I`pwd`/include -L`pwd`/lib -L`pwd`/bin"
	export LDFLAGS=$CFLAGS
	export PKG_CONFIG_PATH=`pwd`/lib/pkg-config

# Do the separate instructions for rubberband, portaudio, etc...
ln -s libportaudio-2.dll libportaudio.dll
